My Function

Multiplies 2 input numbers together and prints result in the console.

Multiply

            let result;

            function multiply(num1, num2) {
                result = num1 * num2;
                console.log(result);
            };

            multiply(6, 2);

Function Expression

            let result;
            
            const multiply = function(num1, num2) {
                result = num1 * num2;
                console.log(result);
            };

            multiply(6, 2);

Arrow Function

            let result;
            
            const multiply = (num1, num2) => {
                result = num1 * num2;
                console.log(result);
            };
